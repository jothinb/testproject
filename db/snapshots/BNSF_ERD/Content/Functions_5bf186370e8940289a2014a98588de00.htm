          /* First off open a cursor for any units that match the booking */
          IF NOT MAU_VIEW%isopen THEN
              OPEN MAU_VIEW;
          END IF;


          LOOP

             FETCH MAU_VIEW INTO t_mau_init, t_mau_num, mau_mounted_stacked, t_mau_x;

             /* If nothing found, then break the loop and set status to EOF */
             IF MAU_VIEW%notfound THEN
                  CLOSE MAU_VIEW;
                  ROLLBACK;
                      IF first_time = 1 THEN

                      ELSE

                      END IF;

             /* start by returning immediately if we find one with wheels */

                mau_num := t_mau_num;
                mau_x := t_mau_x;

                      l_num := t_mau_num;

             IF( l_count < o_count   OR (l_count = o_count AND t_mau_x < mau_x ) ) THEN

                mau_init := t_mau_init;
                mau_num := t_mau_num;
                mau_x := t_mau_x;
             END IF;

                first_time := 0;

    END;

    o_unit_num := mau_num;

RETURN (status);
